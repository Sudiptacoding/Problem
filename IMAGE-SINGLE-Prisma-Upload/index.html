# src এবং uploads ফোল্ডার একই ডিরেক্টারিতে থাকবে src--- uploads--- # Router ---- app.use('/', router); ------------------------------------ import express from 'express'; import upload from './upload'; import { getAllBlogs, deleteBlog, updateBlog, createBlog, } from './createBlog'; const router = express.Router(); router.get('/blogs', getAllBlogs); router.post('/blogs', upload.single('image'), createBlog); router.put('/blogs/:id', upload.single('image'), updateBlog); router.delete('/blogs/:id', deleteBlog); export default router; # Upload Middleware ---- import multer from 'multer'; import fs from 'fs'; const dir = './uploads'; if (!fs.existsSync(dir)) { fs.mkdirSync(dir, { recursive: true }); } const storage = multer.diskStorage({ destination: (req, file, cb) => { cb(null, dir); // `uploads` ফোল্ডার }, filename: (req, file, cb) => { cb(null, Date.now() + '-' + file.originalname); // ফাইলের নামের সাথে টিমস্ট্যাম্প যোগ করা }, }); const upload = multer({ storage: storage }); export default upload; # All CRUD Function ---- import { Request, Response } from 'express'; import fs from 'fs'; import path from 'path'; import { prismaClient } from '..'; export const createBlog = async (req: Request, res: Response): Promise => { try { const { title, description } = req.body; // ফাইল চেক করা if (!req.file) { return res.status(400).json({ message: 'Image file is required', }); } // ফাইল চেক করা if (!req.body) { return res.status(400).json({ message: 'Image file is required', }); } const image = req.file.path; // ফাইলের পাথটি নিন // Blog ডাটাবেজ এ নতুন এন্ট্রি তৈরি করুন const blog = await prismaClient.blog.create({ data: { title, description, image, }, }); return res.status(400).json({ message: 'Blog created successfully', data: blog, }); } catch (error) { return res.status(400).json({ message: 'Could not create blog', error: "not found data", }); } }; // Get all blogs export const getAllBlogs = async (req: Request, res: Response): Promise => { try { const blogs = await prismaClient.blog.findMany(); return res.status(400).json({ message: 'Blogs retrieved successfully', data: blogs, }); } catch (error) { return res.status(400).json({ message: 'Could not retrieve blogs', error: 'data NOT_FOUND', }); } }; // Update a blog export const updateBlog = async (req: Request, res: Response): Promise => { try { const { id } = req.params; const { title, description } = req.body; // Fetch existing blog to check if it has an image const existingBlog = await prismaClient.blog.findUnique({ where: { id: +id } }); if (!existingBlog) { return res.status(400).json({ message: 'Blog not found' }); } let imagePath = existingBlog.image; // If a new image is uploaded, replace the old image if (req.file) { imagePath = req.file.path; // Delete old image if it exists if (existingBlog.image && fs.existsSync(existingBlog.image)) { fs.unlinkSync(existingBlog.image); } } // Update blog in the database const updatedBlog = await prismaClient.blog.update({ where: { id: +id }, data: { title, description, image: imagePath }, }); return res.status(400).json({ message: 'Blog updated successfully', data: updatedBlog, }); } catch (error) { return res.status(400).json({ message: 'Could not update blog', error: 'data faild', }); } }; // Delete a blog export const deleteBlog = async (req: Request, res: Response): Promise => { try { const { id } = req.params; // Fetch blog to get image path const blog = await prismaClient.blog.findUnique({ where: { id: +id } }); if (!blog) { return res.status(400).json({ message: 'Blog not found' }); } // Delete image from the filesystem if it exists if (blog.image && fs.existsSync(blog.image)) { fs.unlinkSync(blog.image); } // Delete blog from the database await prismaClient.blog.delete({ where: { id: +id } }); return res.status(200).json({ message: 'Blog deleted successfully', }); } catch (error) { return res.status(404).json({ message: 'Could not delete blog', error: "Something rong", }); } };
